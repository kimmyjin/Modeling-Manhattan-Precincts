#plot(manh,axes=TRUE)
library(raster)
ext = st_bbox(manh) %>% .[c("xmin","xmax","ymin","ymax")] %>% extent()
r = raster(ext, ncol=700, nrow=2100)
r = rasterize(as(manh,"Spatial"),r)
plot(r)

Get prediction locations
